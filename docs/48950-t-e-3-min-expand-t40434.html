-6r3c2 & -6r4c2 => r3n6{c2 c1}- r4n6{c2 c4}- r4n8{c4 c7} => -6r5c1 & -8r5c7

-6r3c2 & -6r4c2 => r3n6{c2 c1}- b4n6{c1 c3}- c4n6{r5 r4}- r4n8{c4 c7} => -6r5c1 & -8r5c7
